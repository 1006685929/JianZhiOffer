package sy181124;

import java.util.Stack;

public class 从尾到头打印链表
{

	public static void main(String[] args)
	{
		Node n1 = new Node(3);
		Node n2 = new Node(6);
		Node n3 = new Node(2);
		Node n4 = new Node(1);
		
		n1.next=n2;
		n2.next=n3;
		n3.next=n4;
	//	printreserveNode(n1);
		printdigui(n1);
		
		

	}
	
	/**
	 * 栈实现
	 * @param node
	 */
	public static void printreserveNode(Node node)
	{
		Stack<Node> stack=new Stack<Node>();
		while(node!=null)
		{
			stack.push(node);
			node=node.next;
		}
		while(!stack.isEmpty())
		{
			System.out.println(stack.pop().val);
		}
	}
	
	/**
	 * 递归实现
	 * @param node
	 */
	public static void printdigui(Node node)
	{
		if(node==null)
			return;
		else {
			printdigui(node.next);
			System.out.println(node.val);
		}
	}

}

class Node
{
	Node next=null;
	int val;
	public Node(int val)
	{
		this.val=val;
	}
}
